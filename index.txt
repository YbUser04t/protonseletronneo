// Function to play a selected music file
function playMusic(fileName) {
    // Get references to the audio player and its source element
    const audioPlayer = document.getElementById('audioPlayer');
    const audioSource = document.getElementById('audioSource');
    const caption = document.getElementById('caption');

    // Update the source of the audio player to the selected music file
    audioSource.src = './' + fileName;
    // Reload the player to apply the new source
    audioPlayer.load();
    // Start playing the music
    audioPlayer.play();

    // Update the caption to show the currently playing music
    caption.textContent = `Legenda: Reproduzindo ${fileName}`;
}

// Example of QR code generation logic (if dynamic QR codes are needed)
function generateQRCode(contactNumber) {
    // Base URL for the QR code generator API
    const baseURL = 'https://api.qrserver.com/v1/create-qr-code/';
    // Build the complete URL with the contact number
    const qrCodeURL = `${baseURL}?data=tel:${contactNumber}&size=200x200`;
    
    // Update the QR code image source dynamically
    const qrCodeImg = document.getElementById('qrCodeImg');
    qrCodeImg.src = qrCodeURL;
}
